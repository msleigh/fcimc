25subroutine init_random_seed(seed1)
26
27 ! Seed the random number generator.
28 ! If seed1 = 0 then set seed1 randomly using the system clock.
29 ! This will give different sequences of random numbers
30 ! from different runs, so results are not reproducible.
31 ! If seed1 > 0 then results will be reproducible since calling this
32 ! twice with the same seed will initialize the random
33 ! number generator so the same sequence is generated.
34 ! Once seed1 is set, set the other elements of the seed array by adding
35 ! multiples of 37 as suggested in the documentation.
36 ! The length of the seed array is determined by calling
37 !
38 integer, intent(inout) :: seed1
39
40 integer :: nseed, clock, i
41 integer, dimension(:), allocatable :: seed
42
43 call random_seed(size = nseed) ! determine how many numbers needed to seed
44 allocate(seed(nseed))
45
46 if (seed1 == 0) then
47 ! randomize the seed: not repeatable
48 call system_clock(count = clock)
49 seed1 = clock
50 endif
51
52 do i=1,nseed
53 seed(i) = seed1 + 37*(i-1)
54 enddo
55
56 call random_seed(put = seed) ! seed the generator
57 deallocate(seed)
58
59end subroutine init_random_seed
