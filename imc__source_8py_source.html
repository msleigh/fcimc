17def imc_get_energy_sources():
18 """Get energy source terms for surface and mesh."""
19 # Left-hand boundary is a black-body emitter at constant temperature T0
20 # (Energy radiatied per unit area = sigma.T0**4 (sigma = S-B constant)
21 e_surf = phys.sb * bcon.T0 ** 4 * time.dt
22
23 # Emission source term
24 e_body = np.zeros(mesh.ncells) # Energy emitted per cell per time-step
25 e_body[:] = (
26 mesh.fleck[:]
27 * mesh.sigma_p[:]
28 * phys.a
29 * phys.c
30 * mesh.temp[:] ** 4
31 * mesh.dx
32 * time.dt
33 )
34
35 # Total energy emitted
36 e_total = e_surf + sum(e_body[:])
37
38 print("\nEnergy radiated in timestep:")
39 print("\nIn total:")
40 print("{:24.16E}".format(e_total))
41 print("\nBy boundary condition:")
42 print("{:24.16E}".format(e_surf))
43 print("\nBy mesh:")
44 print(e_body)
45
46 return e_surf, e_body, e_total

49def imc_get_emission_probabilities(e_surf, e_body, e_total):
50 """Convert energy source terms to particle emission probabilities."""
51 p_surf = e_surf / e_total
52 # Probability of each cell _given that the particle is from the mesh not the
53 # surface_
54 p_body = np.zeros(mesh.ncells)
55 p_body[:] = np.cumsum(e_body[:]) / sum(e_body[:])
56
57 print("\nProbabilities:")
58 print("\nIn total")
59 print("{:16.8E}".format(p_surf + sum(p_body[:])))
60 print("\nOf boundary particle")
61 print("{:16.8E}".format(p_surf))
62 print("\nOf mesh particle (cumulative)")
63 print(p_body[:])
64
65 return p_surf, p_body

68def imc_get_source_particle_numbers(p_surf, p_body):
69 """Calculate number of source particles to create at surface / throughout mesh."""
70 n_census = part.n_census
71 n_input = part.n_input
72 n_max = part.n_max

74 # Determine total number of particles to source this time-step
75 n_source = n_input
76 if (n_source + n_census) > n_max:
77 n_source = n_max - n_census - mesh.ncells - 1
78
79 print("\nSourcing {:8d} particles this timestep".format(n_source))
80 print("(User requested {:8d} per timestep)".format(n_input))
81
82 n_surf = 0
83 n_body = np.zeros(mesh.ncells, dtype=np.uint64)
84
85 # Calculate the number of particles emitted by the surface souce and each mesh cell
86 for _ in range(n_source):

88 n_surf += 1

91 for icell in range(mesh.ncells):

102def imc_source_particles(e_surf, n_surf, e_body, n_body):
103 """For known energy distribution (surface and mesh), create source particles."""
104 # Create the surface-source particles
105 nrg = e_surf / float(n_surf)
106 startnrg = nrg
107 for _ in range(n_surf):
108 origin = -1
109 xpos = 0.0
110 muu = math.sqrt(ran.random()) # Corresponds to f(mu) = 2mu
111 ttt = time.time + ran.random() * time.dt
112 frq = bcon.T0 * sample_planck_spectrum() / phys.h
113 part.prop.append(
114 [origin, ttt, 0, xpos, muu, frq, nrg, startnrg]
115 ) # Add this ptcl to the global list
116
117 # Create the body-source particles
118 for icell in range(mesh.ncells):
119 if n_body[icell] <= 0:
120 continue
121 nrg = e_body[icell] / float(n_body[icell])
122 startnrg = nrg
123 for _ in range(n_body[icell]):
124 origin = icell
125 xpos = mesh.nodepos[icell] + ran.random() * mesh.dx
126 muu = 1.0 - 2.0 * ran.random()
127 ttt = time.time + ran.random() * time.dt
128 frq = mesh.temp[icell] * abs(math.log(ran.random())) / phys.h
129 # Add this ptcl to the global list
130 part.prop.append([origin, ttt, icell, xpos, muu, frq, nrg, startnrg])

134 """
135 Source new IMC particles.
136
137 This routine calculates the energy sources for
138 the (a) left-hand boundary (which is currently held at a constant
139 temperature, T0), and (b) the computational cells, as well as the overall
140 total for the time-step. These are then converted into particle emission
141 probabilities. The number of particles to source in this time-step is
142 determined (ensuring that the total number in the system does not exceed
143 some pre-defined maximum), and then these are attributed either to the
144 boundary or to one of the mesh cells, according to the probabilities
145 calculated earlier. The particles are then created.
146 """
147 print("\n" + "-" * 79)
148 print("Source step ({:4d})".format(time.step))
149 print("-" * 79)

151 # Determine probability of particles belonging to sources
152 # -------------------------------------------------------
